from PIL import Image
import requests

import tempfile

import os

url = 'https://files.catbox.moe/vdzsjq.jpg'

f"border-image: url({Path(f'{temp.name}.bmp').as_posix()}) 0 0 0 0 stretch stretch;"

os.remove(Path(f'{temp.name}.bmp').as_posix())
